Javascript 单例模式

单例就是保证一个类只有一个实例，实现的方法一般是先判断实例存在与否，如果存在直接返回，如果不存在就创建了再返回，这就确保了一个类只有一个实例对象。在JavaScript里，实现单例的方式有很多种，其中最简单的一个方式是使用对象字面量的方法，其字面量里可以包含大量的属性和方法。

单例模式主要应用于系统间各种模块的通信协调上。

单例模式实现示例

            var Singleton = (function() {
                
                var instantiated;
                
                //初始化 返回公有方法（可访问私有方法） 和变量
                function init() {
                    return {
                        publicMethod : function() {
                            privateMethod();
                        },
                        publicVar : "this is the public variable."
                    };
                }

                //私有变量和方法
                var privateVar = "private variable";
                function privateMethod() {
                    console.log(privateVar);
                }

                return {
                    getInstance : function() {
                        if(!instantiated){
                            instantiated = init();
                        }
                        return instantiated;
                    }
                }
                
            })();
            
            /*调用公有的方法来获取实例:*/
            Singleton.getInstance().publicMethod();
